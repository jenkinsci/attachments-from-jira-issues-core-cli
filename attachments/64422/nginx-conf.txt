# configuration file /etc/nginx/conf.d/jenkins.conf:
# Define the upstream for your first Jenkins instance
upstream jenkins_smena {
    server 127.0.0.1:8080 fail_timeout=0; # Assuming Jenkins is running on 127.0.0.1:8080
}

# Required for Jenkins websocket agents
map $http_upgrade $connection_upgrade {
  default upgrade;
  '' close;
}

server {
    listen 80;
    server_name smena;

    location ~ ^/(static|adjuncts)/ {
        # Proxy these requests to Jenkins, ensuring the /jenkins/ prefix is added
        proxy_pass http://jenkins_smena/jenkins$request_uri;

        # Standard proxy headers (copied from /jenkins/ block for consistency)
        proxy_set_header   Connection        $connection_upgrade; # From map
        proxy_set_header   Upgrade           $http_upgrade;       # From map

        proxy_set_header   Host              $http_host;
        proxy_set_header   X-Real-IP         $remote_addr;
        proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_set_header   X-Forwarded-Prefix /jenkins; # Critical for Jenkins to know its prefix

        proxy_max_temp_file_size 0; # Prevent large files from being written to disk unnecessarily
        proxy_read_timeout 90;      # Increase if you have long build logs/downloads
        client_max_body_size 1024m; # Allow large uploads
        proxy_http_version 1.1; # Required for Upgrade header for WebSockets
    }

    # This is the dedicated location block for your Jenkins instance at /jenkins/
    # This will handle requests like http://smena/jenkins/
    location /jenkins/ {
        sendfile off; # Important for proxying

        # CRITICAL: This line tells Nginx to proxy requests for /jenkins/
        # to the backend *including* the /jenkins/ path.
        # So, http://smena/jenkins/some/path becomes http://127.0.0.1:8080/jenkins/some/path
        proxy_pass         http://jenkins_smena/jenkins/;

        # This helps Jenkins rewrite its internal redirects (e.g., /login)
        # to include the /jenkins/ prefix for the client browser.
        proxy_redirect     default;

        proxy_http_version 1.1;

        # Required for Jenkins websocket agents (e.g., console output during builds)
        proxy_set_header   Connection        $connection_upgrade;
        proxy_set_header   Upgrade           $http_upgrade;

        # Standard proxy headers
        proxy_set_header   Host              $http_host;
        proxy_set_header   X-Real-IP         $remote_addr;
        proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_set_header   X-Forwarded-Prefix /jenkins;

        proxy_max_temp_file_size 0; # Prevent large files from being written to disk unnecessarily
        proxy_read_timeout 90;      # Increase if you have long build logs/downloads
        client_max_body_size 1024m; # Allow large uploads
    }

    # IMPORTANT: The following specific 'location' blocks (like /static/ and /userContent)
    # that were previously *outside* your 'location /' block, will now only apply
    # to requests that DO NOT start with /jenkins/.
    #
    # Since Jenkins (when configured with X-Forwarded-Prefix /jenkins)
    # should generate all its internal links with the /jenkins/ prefix
    # (e.g., /jenkins/static/... or /jenkins/userContent),
    # these requests will be handled by the 'location /jenkins/' block above.
    #
    # If, however, you have other applications or specific static content
    # you want to serve directly from Nginx at the root, you can define them here.

    # Example: If you have general static files at the root, or for other apps.
    # Requests like http://smena/index.html or http://smena/some_image.png
    # that are NOT for Jenkins will be handled here.
    location / {
        root /usr/share/nginx/html; # This is likely your default static file root
        index index.html index.htm; # Default index files for this root
        try_files $uri $uri/ =404;
    }
}
